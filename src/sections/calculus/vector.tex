Let $\vec{r}(t) = x(t) \vec{i} + y(t) \vec{j} + z(t) \vec{k}$ be a vector function.

\begin{multicols}{2}
    $\bm{\vec{r}'(t)}$ is the \textbf{Tangent Vector}.

    $\bm{\vec{T}(t)}$ is the \textbf{Unit Tangent Vector} where 
    $\vec{T}(t) = \frac{\vec{r}'(t)}{\magnitude{\vec{r}'(t)}}$.
    
    $\bm{\vec{N}(t)}$ is the \textbf{Unit Normal Vector} where 
    $\vec{N}(t) = \frac{\vec{T}'(t)}{\magnitude{\vec{T}'(t)}}$.
    
    $\bm{\vec{B}(t)}$ is the \textbf{Binormal Vector} where
    $\vec{B}(t) = \vec{T}(t) \times \vec{N}(t)$.
\end{multicols}

Define the operator
\(
    \bm{\nabla} = \frac{\partial}{\partial x} \vec{i}
    + \frac{\partial}{\partial y} \vec{j}
    + \frac{\partial}{\partial z} \vec{k}
\) where \(
    \nabla f(x,y,z) = \frac{\partial f}{\partial x} \vec{i}
    + \frac{\partial f}{\partial y} \vec{j}
    + \frac{\partial f}{\partial z} \vec{k}
\).

$\nabla f$ is a vector with the gradient vectors of $f$.


Let $\vec{F}(x,y,z) = P(x,y,z) \vec{i} + Q(x,y,z) \vec{j} + R(x,y,z) \vec{k}$ be a vector field.

$\vec{F}$ is \textbf{Conservative} $\iff \lis f$ such that $\vec{F} = \nabla f$,
where $f$ is the \textbf{Potential Function} of $\vec{F}$.

\subsection{Line Integral}
    Let $\vec{F}$ be a continuous vector field with domain $D$.
    % $\int_C \vec{F} \dotp d\vec{r}$ is \textbf{Independent of Path} $\iff
    Let $C$ be a curve.
    % $C$ is \textbf{Closed}
    % $C$ is \textbf{Simple} if it doesn't cross itself.
    Let $f$ be a function of $x_1, x_2, \dots, x_n$.

    The \textbf{Line Integral} along $C$ of
    \(
        \begin{cases}
            f \text{ is }
                \int_C f(x_1,\hdots,x_n) ds 
                = \int_a^b f(x_1(t),\hdots,x_n(t)) \magnitude{\vec{r}' (t)} dt 
            \\
            f \text{ respect to $x_k$ is }
                \int_C f(x_1,\hdots,x_n) dx_k 
                = \int_a^b f(x_1(t),\hdots,x_n(t)) x_k'(t) dt 
            \\
            \vec{F} \text{ is }
                \int_C \vec{F} \dotp d\vec{r}
                = \int_C \vec{F} \dotp \vec{T} ds
                = \int_a^b \vec{F}(\vec{r}(t)) \dotp \vec{r}'(t)dt
        \end{cases}
    \)

    \textbf{Fundamental Theorem of Calculus (for Line Integrals):}\\
    If $C$ is a Smooth curve $\vec{r}(t)$ 
    and $\nabla f$ is continuous on $C$, then
    \(
        \int_C \nabla f \dotp d\vec{r}
        = f(\vec{r} (b)) - f(\vec{r} (a))
    \)

    Let $C$ positively oriented smooth simple closed curve,
